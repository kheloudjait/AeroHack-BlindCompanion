import cv2
from ultralytics import YOLO
from gtts import gTTS
import os
import threading
import time

# --- CONFIGURATION ---
# On charge le modèle d'IA qui reconnaît 80 objets (YOLOv8)
model = YOLO('yolov8n.pt') 

# Variable pour ne pas que l'IA parle toutes les millisecondes
derniere_alerte = 0
delai_entre_alertes = 4 # secondes

def parler(texte):
    """Fonction pour transformer le texte en voix sans bloquer l'image"""
    def execution():
        try:
            tts = gTTS(text=texte, lang='fr')
            tts.save("alerte.mp3")
            # Commande Windows pour lire le son (cache la fenêtre noire)
            os.system("start /min mp3play.exe alerte.mp3" if os.name == 'nt' else "afplay alerte.mp3")
            # Si 'start' ouvre un lecteur, utilise 'pip install pygame' pour une lecture plus pro
        except:
            pass
    threading.Thread(target=execution).start()

# --- BOUCLE PRINCIPALE ---
cap = cv2.VideoCapture(0) # Ouvre la webcam

print("Système AeroHack activé... Appuyez sur 'q' pour quitter.")

while cap.isOpened():
    ret, frame = cap.read()
    if not ret:
        break

    # 1. L'IA analyse l'image
    results = model(frame, conf=0.5, verbose=False)
    
    # 2. On récupère les noms des objets détectés
    objets_detectes = []
    for r in results:
        for box in r.boxes:
            nom = model.names[int(box.cls[0])]
            objets_detectes.append(nom)

    # 3. Logique d'alerte pour l'aveugle
    # On définit ce qui est un "obstacle" pour nous
    obstacles = ["chair", "person", "bottle", "table", "backpack"]
    
    maintenant = time.time()
    if maintenant - derniere_alerte > delai_entre_alertes:
        trouve = [o for o in objets_detectes if o in obstacles]
        if trouve:
            objet_principal = trouve[0]
            # Traduction simple pour la démo
            traductions = {"chair": "une chaise", "person": "une personne", "bottle": "une bouteille", "table": "une table"}
            nom_fr = traductions.get(objet_principal, objet_principal)
            
            parler(f"Attention, il y a {nom_fr} devant vous.")
            derniere_alerte = maintenant

    # 4. Affichage visuel (pour les juges)
    # Dessine les boîtes et les noms sur l'écran
    annotated_frame = results[0].plot()
    cv2.imshow("AeroHack - Vision Compagnon", annotated_frame)

    # Quitter avec la touche 'q'
    if cv2.waitKey(1) & 0xFF == ord('q'):
        break

cap.release()
cv2.destroyAllWindows()
